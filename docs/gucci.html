get_precincts <- function(sgId = "20160413") {

    serviceKey = Sys.getenv("DATA_GO_DECODE_KEY"),

    numOfRows = 1000,

    fromJSON() %>% 
    .$getCommonSggCodeList %>% 
    .$item  %>% 
    as_tibble() 
  
  return(precincts)
}
