mst_tbl_geocodes_sample <- 
  read_rds("data/incheon.rds")

mst_tbl_geocodes_sample %>%
  mutate(popup = paste0("<b>", 회사명, "</b>", "<br>",
                        주소, "<br>")) %>%
  leaflet() %>% 
  addTiles() %>%
  addProviderTiles(providers$CartoDB.Positron) %>%
  addMarkers(~long, ~lat, 
             popup  = ~popup) %>%
  addLayersControl(overlayGroups = ~회사명,
                   options = layersControlOptions(collapsed = FALSE))
