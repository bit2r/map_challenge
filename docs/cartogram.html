코드
library(tidyverse)
library(rvest)
library(httr)
library(sf)
library(geogrid)
sf_use_s2(FALSE)

precinct_sf <- st_read("data/2020_21_elec_253.json")

sido_sf <- precinct_sf |> 
  # filter(SGG_1 == "부산") |> 
  calculate_grid(grid_type="hexagonal", learning_rate = 0.05, seed=20)

plot(sido_sf)

sido_hex <- assign_polygons(precinct_sf, sido_sf)
