1.1 국회의원 선거 코드

코드 
library(tidyverse)
library(rvest)
library(gt)
library(gtExtras)
library(rvest)

# iconv(names(krvote::code_election), from = "EUC-KR", to = "UTF-8")

code_election <- krvote::code_election

names(code_election) <- c("선거코드", "선거명", "선거구분")

legislator_code <- code_election |> 
  filter(str_detect(선거명, "국회의원")) |> 
  pull(선거코드) |> 
  unique()
